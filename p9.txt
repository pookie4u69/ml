import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_olivetti_faces
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import GaussianNB
from sklearn.metrics import accuracy_score, classification_report

faces = fetch_olivetti_faces()

X_faces_data = faces.data
y_faces_target = faces.target

X_faces_train, X_faces_test, y_faces_train, y_faces_test = train_test_split(X_faces_data, y_faces_target, test_size=0.2, random_state=42, stratify=y_faces_target)
gnb = GaussianNB()

gnb.fit(X_faces_train, y_faces_train)

y_faces_pred = gnb.predict(X_faces_test)

accuracy=accuracy_score(y_faces_test, y_faces_pred)
print("Accuracy: of NB on OFD : {:.2f}%".format(accuracy * 100))
print("\nclassification")

fig, axes = plt.subplots(2,5, figsize=(12,5))
for i, ax in enumerate(axes.flat):
    ax.imshow(X_faces_test[i].reshape(64, 64), cmap='grey')
    ax.set_title(f"Predicted: {y_faces_pred[i]}\nActual: {y_faces_test[i]}")
    ax.axis('off')

plt.tight_layout()
plt.show()
print(classification_report(y_faces_test, y_faces_pred))
