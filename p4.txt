
import pandas as pd

data=pd.read_csv("/content/train_data - Sheet1.csv")
print(data)

positive_examples=data[data['Class']==1]
print(positive_examples)

if positive_examples.empty:
  print("\n No positive data")

hypo=list(positive_examples.iloc[0, :-1])
print(hypo)

for _,example in positive_examples.iloc[1:].iterrows():
  print(f"Examples:{example[:-1]}\n")

  for i in range(len(hypo)):
    if hypo[i]!=example.iloc[i]:
      hypo[i]='?'

  print(f"Hypothesis:{hypo}\n")

print(f"Final Hypothesis after ecample {example.name}:{hypo}\n")
